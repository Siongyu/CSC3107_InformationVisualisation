This data frame has 5 rows and 4 columns which is relatively wide. We can pivot this data frame to make it longer.

ny_mayor_long <-
  pivot_longer(ny_mayor,
    cols = c("de_blasio", "malliotakis", "other"),
    names_to = "candidate",
    values_to = "votes"
  )
